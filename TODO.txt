CREATE TABLE media (
    sorszam INT PRIMARY KEY AUTO_INCREMENT,
    cim VARCHAR(255) NOT NULL,
    beszerzes_datuma DATE NOT NULL,
    tipus ENUM('kazetta', 'DVD') NOT NULL,
    statusz ENUM('szabad', 'kikölcsönzött', 'selejtezett') DEFAULT 'szabad'
);

CREATE TABLE borrowings (
    id INT PRIMARY KEY AUTO_INCREMENT,
    sorszam INT NOT NULL,
    azonosito INT NOT NULL,
    kolcsonzes_datuma DATE NOT NULL,
    visszahozas_datuma DATE DEFAULT NULL,
    FOREIGN KEY (sorszam) REFERENCES media(sorszam),
    FOREIGN KEY (azonosito) REFERENCES customer(azonosito)
);
